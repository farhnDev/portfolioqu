const express = require('express');
const bodyParser = require('body-parser');
const sgMail = require('@sendgrid/mail');
const cors = require('cors');

const app = express();
app.use(bodyParser.json());
app.use(cors());

// Set API key
sgMail.setApiKey('YOUR_SENDGRID_API_KEY');

app.post('/send-email', (req, res) => {
    const { from_name, from_email, subject, message } = req.body;

    const msg = {
        to: 'farhanmaulana1710@gmail.com',  // Replace with your email
        from: from_email,  // Use the user's email address as the sender
        subject: subject,
        text: message,
        html: `<strong>${message}</strong>`,
    };

    sgMail
        .send(msg)
        .then(() => {
            console.log('Email sent');
            res.status(200).send('Email sent');
        })
        .catch((error) => {
            console.error(error);
            res.status(500).send('Error sending email');
        });
});

const PORT = process.env.PORT || 5000;
app.listen(PORT, () => {
    console.log(`Server is running on port ${PORT}`);
});

import React, { useState } from 'react';
import axios from 'axios';

export default function Contacts() {
    const [formData, setFormData] = useState({
        from_name: '',
        from_email: '',
        subject: '',
        message: ''
    });

    const [statusMessage, setStatusMessage] = useState('');

    const handleChange = (e) => {
        const { name, value } = e.target;
        setFormData({
            ...formData,
            [name]: value
        });
    };

    const handleSubmit = (e) => {
        e.preventDefault();

        axios.post('http://localhost:5000/send-email', formData)
            .then((response) => {
                console.log('SUCCESS!', response.data);
                setStatusMessage('Pesan berhasil dikirim!');
                setFormData({
                    from_name: '',
                    from_email: '',
                    subject: '',
                    message: ''
                });
            })
            .catch((error) => {
                console.log('FAILED...', error);
                setStatusMessage('Pesan gagal dikirim. Silakan coba lagi.');
            });
    };
